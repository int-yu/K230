# K230 Wi-Fi RTSP Display Streaming Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add opt-in, fail-open Wi-Fi RTSP streaming of the final annotated CanMV display without changing existing detector, UART, capture, or default camera behavior.

**Architecture:** A new `WifiRtspService` lazily loads the K230 `network` and official `libs.WBCRtsp` modules, connects to a 2.4 GHz hotspot, and starts the firmware-provided WBC-to-hardware-H.264 RTSP pipeline. `CameraIO` owns the optional service lifecycle after Display and MediaManager initialization; failures are contained unless the caller explicitly requests strict behavior.

**Tech Stack:** CanMV K230 MicroPython, `network.WLAN`, official `libs.WBCRtsp`, hardware VENC H.264, CPython pytest with dependency injection and fake media modules.

## Global Constraints

- `WIFI_RTSP_ENABLED` defaults to `False`; all existing programs retain their current behavior until explicitly enabled.
- `WIFI_RTSP_REQUIRED` defaults to `False`; missing credentials, firmware modules, hotspot access, DHCP, WBC, or RTSP must not stop existing visual, UART, capture, or IDE Preview behavior.
- Keep the selected Display output unchanged; do not disable `Display.VIRT(..., to_ide=True)` during RTSP initialization.
- Use the official fixed WBC RTSP settings: H.264, no audio, about 2048 kbps, port `8554`, session `test`, and current Display dimensions.
- Never call official `WBCRtsp.stop()` unless `WBCRtsp.start()` returned successfully, because the official stop path waits for a worker thread.
- Do not modify detector algorithms, UART frame formats, Bluetooth UART behavior, capture behavior, camera resolution, or display resolution.
- Add no third-party board dependency such as openRTSP or LIVE555; VLC/ffplay/openRTSP are optional computer-side clients only.
- Store real hotspot credentials only in ignored `wifi_secrets.py`; commit only `wifi_secrets.example.py`.

```python
"""K230 Wi-Fi STA 与官方 WBC RTSP 的 fail-safe 生命周期封装。"""

import time


class WifiRtspService:
    """连接热点并把最终 Display 画面通过官方 WBCRtsp 推流。"""

    def __init__(self, ssid, password, connect_timeout_s=15,
                 wlan_factory=None, wbc=None, time_module=None):
        if not isinstance(ssid, str) or not ssid.strip():
            raise ValueError("WIFI_SSID 不能为空")
        if not isinstance(password, str) or not password:
            raise ValueError("WIFI_PASSWORD 不能为空")
        if connect_timeout_s <= 0:
            raise ValueError("connect_timeout_s 必须大于 0")
        self.ssid = ssid.strip()
        self.password = password
        self.connect_timeout_s = float(connect_timeout_s)
        self._wlan_factory = wlan_factory
        self._wbc = wbc
        self._time = time_module or time
        self._wlan = None
        self._wbc_started = False
        self.active = False
        self.ip_address = None
        self.rtsp_url = None
        self.last_error = None

    def initialize(self, width, height):
        if self.active:
            return self
        self.last_error = None
        try:
            self._wlan = self._create_wlan()
            active_method = getattr(self._wlan, "active", None)
            if active_method is not None:
                active_method(True)
            self._wlan.connect(self.ssid, self.password)
            self._wait_for_connection()
            network_config = self._wlan.ifconfig()
            ip_address = network_config[0]
            if not ip_address or ip_address == "0.0.0.0":
                raise RuntimeError("热点已连接但 DHCP 未取得有效 IP")
            wbc = self._get_wbc()
            wbc.configure(int(width), int(height))
            wbc.start()
            self._wbc_started = True
            self.ip_address = str(ip_address)
            self.rtsp_url = self._resolve_rtsp_url()
            self.active = True
            return self
        except Exception as error:
            message = "Wi-Fi RTSP 启动失败: {}".format(error)
            self.deinitialize()
            self.last_error = message
            raise RuntimeError(message)

    def deinitialize(self):
        if self._wbc_started:
            try:
                self._wbc.stop()
            except Exception as error:
                if self.last_error is None:
                    self.last_error = "停止 WBC RTSP 失败: {}".format(error)
            self._wbc_started = False
        if self._wlan is not None:
            try:
                self._wlan.disconnect()
            except Exception:
                pass
            active_method = getattr(self._wlan, "active", None)
            if active_method is not None:
                try:
                    active_method(False)
                except Exception:
                    pass
            self._wlan = None
        self.active = False
        self.ip_address = None
        self.rtsp_url = None

    def _create_wlan(self):
        if self._wlan_factory is not None:
            return self._wlan_factory()
        try:
            import network
        except ImportError:
            raise RuntimeError(
                "固件缺少 network 模块，请升级到支持 WLAN 的 CanMV 固件"
            )
        if not hasattr(network, "WLAN") or not hasattr(network, "STA_IF"):
            raise RuntimeError("当前固件的 network 模块不支持 WLAN STA")
        return network.WLAN(network.STA_IF)

    def _get_wbc(self):
        if self._wbc is not None:
            return self._wbc
        try:
            from libs.WBCRtsp import WBCRtsp
        except ImportError:
            raise RuntimeError(
                "固件缺少 libs.WBCRtsp，请升级 CanMV 固件或上传官方库"
            )
        self._wbc = WBCRtsp
        return self._wbc

    def _wait_for_connection(self):
        start_ms = self._ticks_ms()
        timeout_ms = int(self.connect_timeout_s * 1000)
        while not self._wlan.isconnected():
            if self._ticks_diff(self._ticks_ms(), start_ms) >= timeout_ms:
                raise RuntimeError(
                    "连接热点超时（{} 秒）".format(self.connect_timeout_s)
                )
            self._sleep_ms(100)

    def _resolve_rtsp_url(self):
        server = getattr(self._wbc, "rtspserver", None)
        getter = getattr(server, "get_rtsp_url", None)
        if getter is not None:
            try:
                url = getter()
                if url:
                    return str(url)
            except Exception:
                pass
        return "rtsp://{}:8554/test".format(self.ip_address)

    def _ticks_ms(self):
        ticks_ms = getattr(self._time, "ticks_ms", None)
        if ticks_ms is not None:
            return ticks_ms()
        return int(self._time.time() * 1000)

    def _ticks_diff(self, current, start):
        ticks_diff = getattr(self._time, "ticks_diff", None)
        if ticks_diff is not None:
            return ticks_diff(current, start)
        return current - start

    def _sleep_ms(self, milliseconds):
        sleep_ms = getattr(self._time, "sleep_ms", None)
        if sleep_ms is not None:
            sleep_ms(milliseconds)
        else:
            self._time.sleep(milliseconds / 1000.0)


def load_wifi_credentials(module_name="wifi_secrets"):
    try:
        module = __import__(module_name)
    except ImportError:
        raise RuntimeError(
            "缺少 {}.py，请复制 wifi_secrets.example.py 并填写热点".format(
                module_name
            )
        )
    ssid = getattr(module, "WIFI_SSID", None)
    password = getattr(module, "WIFI_PASSWORD", None)
    if not isinstance(ssid, str) or not ssid.strip():
        raise RuntimeError("{}.WIFI_SSID 不能为空".format(module_name))
    if not isinstance(password, str) or not password:
        raise RuntimeError("{}.WIFI_PASSWORD 不能为空".format(module_name))
    return ssid.strip(), password


def create_default_wifi_rtsp_service(connect_timeout_s=15):
    ssid, password = load_wifi_credentials()
    return WifiRtspService(
        ssid, password, connect_timeout_s=connect_timeout_s
    )
```

```python
            self._rtsp_error = None
            print("Wi-Fi RTSP 已启动: {}".format(service.rtsp_url))
        except Exception as error:
            if self._rtsp_service is not None:
                try:
                    self._rtsp_service.deinitialize()
                except Exception:
                    pass
            service_error = None
            if self._rtsp_service is not None:
                service_error = getattr(
                    self._rtsp_service, "last_error", None
                )
            self._rtsp_error = service_error or str(error)
            if self.rtsp_required:
                raise
            print("Wi-Fi RTSP 未启用，原功能继续: {}".format(
                self._rtsp_error
            ))
```
